CREATE SEQUENCE SEQ_KMEMBER
INCREMENT 1
MINVALUE 1
START 1
CACHE 1;

CREATE TABLE KMEMBER_ACCOUNT
(
    KMB_SEQ INTEGER PRIMARY KEY DEFAULT NEXTVAL('SEQ_KMEMBER'),
    KMB_EMAIL varchar(40) UNIQUE,
    KMB_STATUS varchar(10) NOT NULL DEFAULT 'NOT EXIST'
);

SELECT * FROM KMEMBER_ACCOUNT;
